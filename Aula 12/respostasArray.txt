Exercício 13

        let vetor = []
        let somaNotas = 0
        let maiorNota = 0
        while(true){
            let notas = +prompt("Digite uma nota")
            if(notas<0){
                break
            }
            vetor.push(notas)
        }
        for(let elemento of vetor){
            somaNotas+=elemento
            if(elemento>maiorNota){
                maiorNota=elemento
            }
        }
        let media = somaNotas/vetor.length
        console.log(`Média geral da turma é ${media} e a maior nota é ${maiorNota}`)
